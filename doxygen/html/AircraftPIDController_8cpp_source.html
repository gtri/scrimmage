109  altitude_pid_.set_setpoint(vars_.input(desired_altitude_idx_));
110  double desired_pitch = -altitude_pid_.step(time_->dt(), state_->pos()(2));
111  desired_pitch = clamp(desired_pitch, -max_pitch_, max_pitch_);
112

117 
118  double desired_roll;

121  } else {
