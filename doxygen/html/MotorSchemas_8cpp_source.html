153 bool MotorSchemas::step_autonomy(double t, double dt) {
154  // Run all sub behaviors
155  double vec_w_gain_sum = 0;
156  Eigen::Vector3d vec_w_gain(0, 0, 0);
157 
158  for (motor_schemas::BehaviorBasePtr &behavior : current_behaviors_) {
159  behavior->shapes().clear();
160 
161  // cout << "Behavior: " << behavior->name() << endl;
162 
163  // Execute callbacks for received messages before calling
164  // step_autonomy
165  for (SubscriberBasePtr &sub : behavior->subs()) {
166  for (auto msg : sub->pop_msgs<sc::MessageBase>()) {
167  sub->accept(msg);
168  }
169  }
170  if (!behavior->step_autonomy(time_->t(), time_->dt())) {
171  cout << "MotorSchemas: behavior error" << endl;
172  }
173 
174  // Grab the desired vector and normalize to max_speed if too large
175  Eigen::Vector3d desired_vector = behavior->desired_vector();
176  if (desired_vector.norm() > max_speed_) {
177  desired_vector = desired_vector.normalized() * max_speed_;
178  }
179 
180  if (desired_vector.hasNaN()) {
181  cout << "Behavior error: " << behavior->name()
182  << ", desired vector has NaN" << endl;
183  continue;
184  }
185 
186  // cout << "desired_vector: " << desired_vector << endl;
187  // cout << "gain: " << behavior->gain() << endl;
188  // cout << "desired_vector.norm(): " << desired_vector.norm() << endl;
189 
190  // Keep a running sum of all vectors with gains
191  vec_w_gain += desired_vector * behavior->gain();
192 
193  // Keep a running sum of all gains that have an effective vector
194  Eigen::Vector3d desired_vector_normalized = desired_vector.normalized();
195  if (!desired_vector_normalized.hasNaN()) {
196  vec_w_gain_sum += desired_vector.normalized().norm() * behavior->gain();
197  }
198 
199  if (show_shapes_) {
200  std::for_each(behavior->shapes().begin(),
201  behavior->shapes().end(), [&](auto &s) {
202  this->draw_shape(s);
203  });
204  }
205  behavior->shapes().clear();
206  }
207 
208  Eigen::Vector3d vel_result = vec_w_gain / vec_w_gain_sum;
209 
210  // If the vel_result has a NaN value, just go straight. NaNs occur during
211  // initialization of some behaviors.
212  if (vel_result.hasNaN()) {
213  vel_result = state_->quat() * Eigen::Vector3d::UnitX() * max_speed_;
214  }
215 
217  // Publish the resultant velocity vector or
218  // Convert resultant vector into heading / speed / altitude command:
220  if (pub_vel_vec_) {
221  // Add a lower bound on the z-component of the resulting velocity vector
222  if (add_lower_bound_to_vz_) {
223  vel_result(2) = std::max(vz_lower_bound_, vel_result(2));
224  }
225  vars_.output(output_vel_x_idx_, vel_result(0));
226  vars_.output(output_vel_y_idx_, vel_result(1));
227  vars_.output(output_vel_z_idx_, vel_result(2));
228  } else {
229  double heading = sc::Angles::angle_2pi(atan2(vel_result(1), vel_result(0)));
230  vars_.output(desired_alt_idx_, state_->pos()(2) + vel_result(2));
231  vars_.output(desired_speed_idx_, vel_result.norm());
232  vars_.output(desired_heading_idx_, heading);
233  }
234 
236  // Draw important shapes
238  if (show_shapes_) {
239  // Draw resultant vector:
240  if (line_shape_ == nullptr) {
241  line_shape_ = sc::shape::make_line(
242  state_->pos(), vel_result + state_->pos(),
243  Eigen::Vector3d(255, 255, 0), 0.75);
244  }
245  sc::set(line_shape_->mutable_line()->mutable_start(), state_->pos());
246  sc::set(line_shape_->mutable_line()->mutable_end(), vel_result + state_->pos());
247  draw_shape(line_shape_);
248  }
249  return true;
250 }
