195 void ContactBlobCamera::draw_frustum(const std::vector<scrimmage_proto::ShapePtr>& frustum_shapes, double x_rot, double y_rot, double z_rot) {
196  if (frustum_shapes.size() != 8) {
197  std::cerr << "ContactBlobCamera::draw_frustum: ERROR: input shape vector must be size 8" << std::endl;
198  return;
199  }
200  double sensor_footprint_height = max_detect_range_ * tan(el_thresh_ / 2) * 2;
201  double sensor_footprint_width = max_detect_range_ * tan(az_thresh_ / 2) * 2;
202  Eigen::Vector3d sensor_UL(max_detect_range_, sensor_footprint_width / 2, -sensor_footprint_height / 2);
203  Eigen::Vector3d sensor_UR(max_detect_range_, sensor_footprint_width / 2, sensor_footprint_height / 2);
204  Eigen::Vector3d sensor_LR(max_detect_range_, -sensor_footprint_width / 2, sensor_footprint_height / 2);

206  std::vector<Eigen::Vector3d> sensor_scene_bb = {sensor_UL, sensor_UR, sensor_LR, sensor_LL, sensor_UL};
207 
208  // Rotate into sensor frame
209  auto x_rotation = Eigen::AngleAxisd(x_rot, Eigen::Vector3d::UnitX());
210  auto y_rotation = Eigen::AngleAxisd(y_rot, Eigen::Vector3d::UnitY());
211  auto z_rotation = Eigen::AngleAxisd(z_rot, Eigen::Vector3d::UnitZ());
212  auto A = Eigen::Translation3d(parent_->state_truth()->pos()) * z_rotation * y_rotation * x_rotation;
213  for (size_t i = 0; i < sensor_scene_bb.size(); i++) {
214  sensor_scene_bb[i] = A * sensor_scene_bb[i];
215  }
216 
217  // Draw scene box
218  auto scene_bb_line = std::make_shared<sp::Shape>();
219  scene_bb_line->set_opacity(1.0);
220  scene_bb_line->set_persistent(false);
221  scene_bb_line->set_persist_duration(0.0);
222  sc::set(scene_bb_line->mutable_color(), 0, 255, 0);
223 
224  sc::path_to_lines(sensor_scene_bb, scene_bb_line,
225  std::static_pointer_cast<EntityPlugin>(shared_from_this()));
226  // draw 8 lines for the 4 points
227  const Eigen::Vector3d fov_edge_start = parent_->state_truth()->pos();
228  for (int idx_bb = 0; idx_bb != 4; ++idx_bb) {
229  // box edges are idx 0-3 in frustum, fov edges are 4-7
230  const size_t idx_fov_edge = idx_bb + 4;
231 
232  // set box line

245  scrimmage::set(frustum_shapes.at(idx_fov_edge)->mutable_line()->mutable_end(), box_edge_end);

253 void ContactBlobCamera::contacts_to_bounding_boxes(
254  const scrimmage::State &sensor_frame,
255  scrimmage::ContactMap &contacts,
256  std::shared_ptr<sc::Message<ContactBlobCameraType>> &msg) {
257 
258  for (auto &kv : contacts) {
259  // Filter out (skip) own contact
260  if (kv.second.id().id() == parent_->id().id()) continue;
261 
262  // Filter out contacts out of range, should use RTree, but rtree still
263  // requires querying of ID from contact lists. (TODO)
264  if ((kv.second.state()->pos() - sensor_frame.pos()).norm() >
265  max_detect_range_) {
266  continue;
267  }
268 
269  // Don't "detect" current contact relative to false negative probability
270  double r = parent_->random()->rng_uniform(0.0, 1.0);
271  if (r < fn_prob_) continue;
272 
273  // Transform contact into "camera" coordinate system
274  Eigen::Vector3d rel_pos = sensor_frame.rel_pos_local_frame(kv.second.state()->pos());
275 
276  if (!in_field_of_view(rel_pos)) continue;
277 
278  // Convert 3D relative position to 2d image plane position
279  Eigen::Vector2d raster_center = project_rel_3d_to_2d(rel_pos);
280 
281  double object_radius = kv.second.radius();
282 
283  // Get angle between (vector between camera and object center) and
284  // (vector between object center and line tangent to object circle)
285  double beta = acos(object_radius / rel_pos.norm());
286 
287  // Get vector pointing from contact to camera
288  Eigen::Vector2d v_r(-rel_pos(0), -rel_pos(1));
289 
290  // Normalize to unit vector and Make length equal to object radius
291  v_r = v_r.normalized() * object_radius;
292 
293  // Rotate one direction and add circle's center point
294  Eigen::Vector2d v_r_1 = v_r, v_r_2 = v_r;
295  Eigen::Rotation2D<double> rot1(beta);
296  Eigen::Rotation2D<double> rot2(-beta);
297  v_r_1 = rot1.toRotationMatrix() * v_r_1;
298  v_r_2 = rot2.toRotationMatrix() * v_r_2;
299 
300  // Get 3D relative position of rotated vectors
301  Eigen::Vector3d p1(rel_pos(0) + v_r_1(0),
302  rel_pos(1) + v_r_1(1),
303  rel_pos(2));
304 
305  Eigen::Vector3d p2(rel_pos(0) + v_r_2(0),
306  rel_pos(1) + v_r_2(1),
307  rel_pos(2));
308 
309  // Get 2D position of object boundaries
310  Eigen::Vector2d r1 = project_rel_3d_to_2d(p1);
311  Eigen::Vector2d r2 = project_rel_3d_to_2d(p2);
312 
313  // Calculate image radius using distance between object boundaries
314  double object_img_radius = (r1 - r2).norm() / 2.0;
315 
316  // Add noise to position in 2D image plane
317  raster_center(0) += (*pos_noise_[0])(*gener_);
318  raster_center(1) += (*pos_noise_[1])(*gener_);
319 
320  // Add bounding box to frame around object
321  cv::Rect rect(std::floor(raster_center(0))-std::floor(object_img_radius),
322  std::floor(raster_center(1))-std::floor(object_img_radius),
323  std::floor(object_img_radius*2) + 1,
324  std::floor(object_img_radius*2) + 1);
325 
326  if (object_img_radius > 0) {
327  draw_object_with_bounding_box(msg->data.frame, kv.second.id().id(),
328  rect, raster_center, object_img_radius);
329  }
330 
331  // Collect all bounding boxes for current detected object
332  msg->data.bounding_boxes[kv.second.id().id()].push_back(rect);
333  }

454 Eigen::Vector2d ContactBlobCamera::project_rel_3d_to_2d(Eigen::Vector3d rel_pos) {
455  // 3D to 2D transforms are described here:
456  // http://www.scratchapixel.com/lessons/3d-basic-rendering/computing-pixel-coordinates-of-3d-point/mathematics-computing-2d-coordinates-of-3d-points
457 
458  // Convert relative position of contact into computer vision coordinates
459  // for relative coordinates:
460  Eigen::Vector3d pos(rel_pos(1), rel_pos(2), rel_pos(0));
461 
462  // Transform to screen / canvas space
463  Eigen::Vector2d s(focal_length_ * -pos(0) / pos(2),
464  focal_length_ * pos(1) / pos(2));
465 
466  // Transform to Normalized Device Coordinate (NDC) space
467  Eigen::Vector2d ndc((s(0) + canvas_width_/2.0) / canvas_width_,
468  (s(1) + canvas_height_/2.0) / canvas_height_);
469 
470  // Tranform to raster space
471  Eigen::Vector2d r(ndc(0) * img_width_,
472  (1-ndc(1)) * img_height_);
473  return r;
474 }

476 bool ContactBlobCamera::in_field_of_view(Eigen::Vector3d rel_pos) {
477  double az = atan2(rel_pos(1), rel_pos(0));
478  double norm_xy = sqrt(pow(rel_pos(0), 2) + pow(rel_pos(1), 2));
479  double el = atan2(rel_pos(2), norm_xy);
480  if (std::abs(az) > (az_thresh_ / 2) ||
481  std::abs(el) > (el_thresh_ / 2)) {
482  return false;
483  }
484  return true;
485 }
