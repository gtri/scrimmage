345 void UUV6DOF::model(const vector_t &x, vector_t &dxdt, double t) {
346  // Calculate force from weight in body frame:
347  Eigen::Vector3d gravity_vector(0, 0, +mass_*g_);
348  Eigen::Vector3d F_weight = quat_body_.rotate_reverse(gravity_vector);
349 
350  Eigen::Vector3d buoyancy_vector(0, 0, buoyancy_);
351  Eigen::Vector3d F_buoyancy = quat_body_.rotate_reverse(buoyancy_vector);
352 
353  Eigen::Vector3d F_hydro = F_weight - F_buoyancy;
354  Eigen::Vector3d Moments_hydro = c_g_.cross(F_weight) - c_b_.cross(F_buoyancy);
355 
356  double Xprop = thrust_;
357 
358  double X_ext = F_hydro(0) + Xuu_ * x[U] * std::abs(x[U]) + Xu_dot*x[U_dot]
359  + Xwq*x[W]*x[Q] + Xqq*x[Q]*x[Q] + Xvr*x[V]*x[R] + Xrr*x[R]*x[R] + Xprop;
360 
361  double Y_ext = F_hydro(1) + Yvv_*x[V]*std::abs(x[V]) + Yrr*x[R]*std::abs(x[R])
362  + Yv_dot*x[V_dot] + Yr_dot*x[R_dot] + Yur*x[U]*x[R] + Ywp*x[W]*x[P]
363  + Ypq*x[P]*x[Q] + Yuv*x[U]*x[V] + Yuu_delta_r * pow(x[U], 2) * delta_rudder_;
364 
365  double Z_ext = F_hydro(2) + Zww_*x[W]*std::abs(x[W]) + Zqq*x[Q]*std::abs(x[Q])
366  + Zw_dot*x[W_dot] + Zq_dot*x[Q_dot] + Zuq*x[U]*x[Q] + Zvp*x[V]*x[P]
367  + Zrp*x[R]*x[P] + Zuw*x[U]*x[W] + Zuu_delta_s*pow(x[U], 2) * delta_elevator_;
368 
369  // Roll moment
370  double K_ext = Moments_hydro(0) + Kpp*x[P]*std::abs(x[P]) + Kp_dot*x[P_dot] + Kprop_;
371 
372  // Pitch moment
373  double M_ext = Moments_hydro(1) + Mww_*x[W]*std::abs(x[W]) + Mqq_*x[Q]*std::abs(x[Q])
374  + Mw_dot*x[W_dot] + Mq_dot*x[Q_dot] + Muq*x[U]*x[Q] + Mvp*x[V]*x[P] + Mrp*x[R]*x[P]
375  + Muw*x[U]*x[W] + Muu_delta_s*pow(x[U], 2) * delta_elevator_;
376 
377  // Yaw moment
378  double N_ext = Moments_hydro(2) + Nvv*x[V]*std::abs(x[V]) + Nrr*x[R]*std::abs(x[R])
379  + Nv_dot*x[V_dot] + Nr_dot*x[R_dot] + Nur*x[U]*x[R] + Nwp*x[W]*x[P] + Npq*x[P]*x[Q]
380  + Nuv*x[U]*x[V] + Nuu_delta_r*pow(x[U], 2)*delta_rudder_;
381 
382  Eigen::Matrix<double, 6, 1> forces;
383  forces << X_ext, Y_ext, Z_ext, K_ext, M_ext, N_ext;
384 
385  Eigen::Matrix<double, 6, 1> accelerations = masses_inverse_ * forces;
386 
387  dxdt[U] = accelerations(0, 0);
388  dxdt[V] = accelerations(1, 0);
389  dxdt[W] = accelerations(2, 0);
390  dxdt[P] = accelerations(3, 0);
391  dxdt[Q] = accelerations(4, 0);
392  dxdt[R] = accelerations(5, 0);
393 
394  // Compute quaternion derivatives
395  double lambda = 1 - (pow(x[q0], 2) + pow(x[q1], 2) + pow(x[q2], 2) + pow(x[q3], 2));
396  dxdt[q0] = -0.5 * (x[q1]*x[P] + x[q2]*x[Q] + x[q3]*x[R]) + lambda * x[q0];
397  dxdt[q1] = +0.5 * (x[q0]*x[P] + x[q2]*x[R] - x[q3]*x[Q]) + lambda * x[q1];
398  dxdt[q2] = +0.5 * (x[q0]*x[Q] + x[q3]*x[P] - x[q1]*x[R]) + lambda * x[q2];
399  dxdt[q3] = +0.5 * (x[q0]*x[R] + x[q1]*x[Q] - x[q2]*x[P]) + lambda * x[q3];
400 
401  // Normalize quaternion
402  sc::Quaternion quat(x[q0], x[q1], x[q2], x[q3]);
403  quat.w() = x[q0];
404  quat.x() = x[q1];
405  quat.y() = x[q2];
406  quat.z() = x[q3];
407  quat.normalize();
408 
409  // Integrate local velocities to compute local positions
410  Eigen::Vector3d vel_local(x[U], x[V], x[W]);
411  Eigen::Vector3d vel_world = quat.rotate(vel_local); // rot * vel_local;
412  dxdt[Xw] = vel_world(0);
413  dxdt[Yw] = -vel_world(1); // Due to rotated frame
414  dxdt[Zw] = -vel_world(2); // Due to rotated frame
415

427  dxdt[P_dot] = 0;
