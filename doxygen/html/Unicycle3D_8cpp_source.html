254 void Unicycle3D::model(const vector_t &x , vector_t &dxdt , double t) {
255  dxdt[U] = acceleration_;

257  dxdt[W] = 0;

263  double lambda = 1 - (pow(x[q0], 2) + pow(x[q1], 2) + pow(x[q2], 2) + pow(x[q3], 2));
264  dxdt[q0] = -0.5 * (x[q1]*x[P] + x[q2]*x[Q] + x[q3]*x[R]) + lambda * x[q0];
265  dxdt[q1] = +0.5 * (x[q0]*x[P] + x[q2]*x[R] - x[q3]*x[Q]) + lambda * x[q1];
266  dxdt[q2] = +0.5 * (x[q0]*x[Q] + x[q3]*x[P] - x[q1]*x[R]) + lambda * x[q2];
267  dxdt[q3] = +0.5 * (x[q0]*x[R] + x[q1]*x[Q] - x[q2]*x[P]) + lambda * x[q3];
268 
269  // Local position / velocity to global
270  // Normalize quaternion
271  sc::Quaternion quat(x[q0], x[q1], x[q2], x[q3]);
272  quat.normalize();
273 
274  quat = quat * quat_world_;
275  quat.normalize();
276 
277  // Convert local positions and velocities into global coordinates
278  Eigen::Matrix3d rot = quat.toRotationMatrix();
279 
280  Eigen::Vector3d vel_local(x[U], x[V], x[W]);
281  Eigen::Vector3d vel_world = rot * vel_local;
282  dxdt[Xw] = vel_world(0);
