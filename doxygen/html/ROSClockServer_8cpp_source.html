88 void ROSClockServer::publish_clock_msg(const double& t) {
89  // Add the current simulation time to the system time at the start of the
90  // simulation.
91  std::chrono::time_point<std::chrono::system_clock,
92  std::chrono::duration<double>> sim_time_point =
93  sim_start_time_ + std::chrono::duration<double>(t);
94 
95  // Convert the time_point into a duration that represents double time since
96  // the epoch.
97  std::chrono::duration<double> time_since_epoch = sim_time_point.time_since_epoch();
98  double sim_time = time_since_epoch.count();
99 
100  // Extract the seconds and nanoseconds to construct the ros::Time message
101  uint32_t sec = std::floor(sim_time);
102  uint32_t nsec = (sim_time - static_cast<double>(sec)) * 1e9;
103 
104  // Create the ros Clock message and publish it

106  clock_msg.clock = ros::Time(sec, nsec);
