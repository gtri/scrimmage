92 int VariableIO::add_input_variable(const std::string &var) {
93  // If the variable already exists, return its existing index
94  declared_input_variables_.insert(var);
95  auto it = input_variable_index_.find(var);
96  if (it != input_variable_index_.end()) {
97  return it->second;
98  }
99 
100  // If it doesn't exist, add it to the map and resize input vector
101  int idx = next_input_variable_index_;
102  input_variable_index_[var] = idx;
103  ++next_input_variable_index_;
104  *input_ = Eigen::VectorXd::Zero(input_variable_index_.size());
105  return idx;
106 }

108 int VariableIO::add_output_variable(const std::string &var) {
109  // If the variable already exists, return its existing index
110  declared_output_variables_.insert(var);
111  int idx;
112  auto it = output_variable_index_.find(var);

116  // If it doesn't exist, it will be ignored by the next vector
117  idx = output_variable_index_.size();
118  output_variable_index_[var] = idx;
119  }

146 void VariableIO::output(int i, double x) {
