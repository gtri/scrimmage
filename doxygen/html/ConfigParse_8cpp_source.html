63 void ConfigParse::recursive_params(rx::xml_node<char> *root,
64  const std::map<std::string, std::string> &overrides,
65  std::map<std::string, std::string> &params,
66  const std::string &prev) {
67  // End condition
68  if (root == 0 || root->name() == std::string("")
69  || root->name() == std::string(" ")) {
70  return;
71  }
72 
73  // Process this node
74  std::string name = root->name();
75  if (prev != "") {
76  name = prev + "/" + name;
77  }
78 
79  if (params.count(name) > 0) {
80  // Name already exists, consider this a list. Does the size
81  // parameter exist yet?
82  std::string name_size = name + ":size";
83  int size = 2;
84  if (params.count(name_size) > 0) {
85  // It does exist, grab the current size and increment.
86  size = std::stoi(params[name_size]) + 1;
87  }
88  name = name + "_" + std::to_string(size-1);
89  params[name_size] = std::to_string(size);
90  }
91 
92  if (overrides.count(name)) {
93  params[name] = overrides.at(name);
94  } else {
95  params[name] = root->value();
96  }
97 
98  // Recurse (depth-first)
99  recursive_params(root->first_node(), overrides, params, name);
100 
101  // Recurse (sibling node)
102  recursive_params(root->next_sibling(), overrides, params, prev);
103 }
