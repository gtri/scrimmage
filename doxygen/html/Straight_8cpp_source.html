206 bool Straight::step_autonomy(double t, double dt) {
207  if (gen_ents_) {
208  if (time_->t() > (prev_gen_time_ + 2.0)) {
209  prev_gen_time_ = time_->t();
210 
211  // Create a state for the new entity
212  State s(*state_);
213 
214  // Rotate the heading of the new entity by 90 degrees and offset
215  // the initial position to the left of our current position to
216  // avoid collisions.
217  Eigen::AngleAxisd rot_90_z(M_PI/2.0, Eigen::Vector3d::UnitZ());
218  s.pos() = s.pos() + s.quat() * rot_90_z * (Eigen::Vector3d::UnitX() * 10);
219  s.quat() = rot_90_z * s.quat();
220 
221  // Create the GenerateEntity message
222  auto msg = std::make_shared<Message<scrimmage_msgs::GenerateEntity>>();
223  sc::set(msg->data.mutable_state(), s); // Copy the new state
224 
225  // The entity_tag must match the "tag" XML attribute of the entity
226  // to be generated in the mission file.
227  msg->data.set_entity_tag("gen_straight");
228 
229  // Modify the entity's color
230  auto kv_color = msg->data.add_entity_param();
231  kv_color->set_key("color");
232  kv_color->set_value("255, 255, 0");
233 
234  // Modify the entity's visual model
235  auto kv_visual = msg->data.add_entity_param();
236  kv_visual->set_key("visual_model");
237  kv_visual->set_value("sphere");
238 
239  pub_gen_ents_->publish(msg); // Publish the GenerateEntity message
240  }
241  }
242 
243  if (show_text_label_) {
244  // An example of changing a shape's property
245  if (t > 1.0 && t < (1.0 + dt)) {
246  text_shape_->set_opacity(0.5);
247  text_shape_->mutable_text()->set_text("Goodbye, SCRIMMAGE!");
248  draw_shape(text_shape_);
249  }
250 
251  // An example of removing a shape
252  if (t > 5.0 && t < (5.0 + dt)) {
253  text_shape_->set_persistent(false);
254  draw_shape(text_shape_);
255  }
256  }
257 
258  // Read data from sensors...
259  if (!noisy_state_set_) {
260  noisy_state_ = *state_;
261  }
262 
263  if (boundary_ != nullptr && enable_boundary_control_) {
264  if (!boundary_->contains(noisy_state_.pos())) {
265  // Project goal through center of boundary
266  Eigen::Vector3d center = boundary_->center();
267  center(2) = noisy_state_.pos()(2); // maintain altitude
268  Eigen::Vector3d diff = center - noisy_state_.pos();
269  goal_ = noisy_state_.pos() + diff.normalized() * 1e6;
270  }
271  }
272 
273  Eigen::Vector3d diff = goal_ - noisy_state_.pos();
274  Eigen::Vector3d v = speed_ * diff.normalized();
275 
277  // Convert desired velocity to desired speed, heading, and pitch controls
279  double heading = Angles::angle_2pi(atan2(v(1), v(0)));
280  vars_.output(desired_alt_idx_, goal_(2));
281  vars_.output(desired_speed_idx_, v.norm());
282  vars_.output(desired_heading_idx_, heading);
283 
284  noisy_state_set_ = false;
285  return true;
286 }
