71 bool MOOSNode::OnNewMail(MOOSMSG_LIST &Mail) {
72  MOOSMSG_LIST::iterator q;
73  for (q = Mail.begin(); q != Mail.end(); ++q) {
74  CMOOSMsg &msg = *q;
75  std::string key = msg.GetKey();
76  double dval = msg.GetDouble();
77 
78  if (key == "DESIRED_HEADING") {
79  desired_mutex_.lock();
80  desired_.quat().set(0, 0, sc::Angles::deg2rad(dval));
81  desired_mutex_.unlock();
82  } else if (key == "DESIRED_SPEED") {
83  desired_mutex_.lock();
84  desired_.vel() = Eigen::Vector3d::UnitX()*dval;
85  desired_mutex_.unlock();
86  } else if (key == "DESIRED_DEPTH") {
87  desired_mutex_.lock();
88  desired_.pos() = -Eigen::Vector3d::UnitZ()*dval;
89  desired_mutex_.unlock();

91  if (!deployed_) {
92  Notify("DEPLOY", "true");
93  Notify("RETURN", "false");
94  Notify("MOOS_MANUAL_OVERRIDE", "false");
95  deployed_mutex_.lock();
96  deployed_ = true;
97  deployed_mutex_.unlock();
98  }
99  }
100  }
101  return(true);
