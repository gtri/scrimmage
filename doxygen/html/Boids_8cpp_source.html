98 bool Boids::step_autonomy(double t, double dt) {
99  // Find neighbors that are within field-of-view and within comms range
100  std::vector<ID> rtree_neighbors;
101  rtree_->neighbors_in_range(state_->pos(), rtree_neighbors, comms_range_);
102 
103  // Remove neighbors that are not within field of view
104  for (auto it = rtree_neighbors.begin(); it != rtree_neighbors.end();
105  /* no inc */) {
106 
107  // Ignore own position / id
108  if (it->id() == parent_->id().id()) {
109  it = rtree_neighbors.erase(it);
110  } else if (state_->InFieldOfView(*(*contacts_)[it->id()].state(), fov_az_, fov_el_)) {
111  // The neighbor is "in front"
112  ++it;
113  } else {
114  // The neighbor is "behind." Remove it.
115  it = rtree_neighbors.erase(it);
116  }
117  }
118 
119  // move-to-goal behavior
120  Eigen::Vector3d v_goal = (goal_ - state_->pos()).normalized();
121 
122  // Steer to avoid local neighbors
123  // Align with neighbors
124  // Cohesion: move towards average position of neighbors
125  // (i.e., Find centroid of neighbors)
126  double heading = 0;
127  Eigen::Vector3d align(0, 0, 0);
128  Eigen::Vector3d centroid(0, 0, 0);
129 
130  // Compute repulsion vector from each robot (team/nonteam)
131  std::vector<Eigen::Vector3d> O_team_vecs;
132  std::vector<Eigen::Vector3d> O_nonteam_vecs;
133 
134  for (ID id : rtree_neighbors) {
135  bool is_team = (id.team_id() == parent_->id().team_id());
136 
137  StatePtr other_state = (*contacts_)[id.id()].state();
138 
139  // Calculate vector pointing from own position to other
140  Eigen::Vector3d diff = other_state->pos() - state_->pos();
141  double dist = diff.norm();
142 
143  // Calculate magnitude of repulsion vector
144  double min_range = is_team ? minimum_team_range_ : minimum_nonteam_range_;
145  double O_mag = 0;
146  if (dist > sphere_of_influence_) {
147  O_mag = 0;
148  } else if (min_range < dist && dist <= sphere_of_influence_) {
149  O_mag = (sphere_of_influence_ - dist) /
150  (sphere_of_influence_ - min_range);
151  } else if (dist <= min_range) {
152  O_mag = 1e10;
153  }
154 
155  // Calculate repulsion vector
156  Eigen::Vector3d O_dir = - O_mag * diff.normalized();
157  if (is_team) {
158  O_team_vecs.push_back(O_dir);
159  } else {
160  O_nonteam_vecs.push_back(O_dir);
161  }
162 
163  // Calculate centroid of team members and heading alignment
164  if (is_team) {
165  centroid = centroid + other_state->pos();
166  align += other_state->vel().normalized();
167  heading += other_state->quat().yaw();
168  }
169  }
170 
171  Eigen::Vector3d align_vec(0, 0, 0);
172  if (rtree_neighbors.size() > 0) {
173  centroid = centroid / static_cast<double>(rtree_neighbors.size());
174  align = align / static_cast<double>(rtree_neighbors.size());
175  heading /= static_cast<double>(rtree_neighbors.size());
176  align_vec << cos(heading), sin(heading), 0;
177  }
178 
179  // Make sure alignment vector is well-behaved
180  align = align_vec; // TODO
181  Eigen::Vector3d v_align_normed = align.normalized();
182  double v_align_norm = align.norm();
183  if (v_align_normed.hasNaN()) {
184  v_align_normed = Eigen::Vector3d::Zero();
185  v_align_norm = 0;
186  }
187 
188  // Normalize each team repulsion vector and sum
189  Eigen::Vector3d O_team_vec(0, 0, 0);
190  for (Eigen::Vector3d v : O_team_vecs) {
191  if (v.hasNaN()) {
192  continue; // ignore misbehaved vectors
193  }
194  O_team_vec += v;
195  }
196 
197  // Normalize each nonteam repulsion vector and sum
198  Eigen::Vector3d O_nonteam_vec(0, 0, 0);
199  for (Eigen::Vector3d v : O_nonteam_vecs) {
200  if (v.hasNaN()) {
201  continue; // ignore misbehaved vectors
202  }
203  O_nonteam_vec += v;
204  }
205 
206  // Apply gains to independent behaviors
207  Eigen::Vector3d v_goal_w_gain = v_goal * w_goal_;
208  Eigen::Vector3d O_team_vec_w_gain = O_team_vec * w_avoid_team_;
209  Eigen::Vector3d O_nonteam_vec_w_gain = O_nonteam_vec * w_avoid_nonteam_;
210  Eigen::Vector3d v_centroid_w_gain = (centroid - state_->pos()).normalized() * w_centroid_;
211  Eigen::Vector3d v_align_w_gain = v_align_normed * w_align_;
212 
213  double sum_norms = v_goal_w_gain.norm() + O_team_vec_w_gain.norm() +
214  O_nonteam_vec_w_gain.norm() + v_centroid_w_gain.norm() +
215  v_align_norm;
216 
217  Eigen::Vector3d v_sum = (v_goal_w_gain + O_team_vec_w_gain +
218  O_nonteam_vec_w_gain + v_centroid_w_gain +
219  v_align_w_gain) / sum_norms;
220 
221  // Scale velocity to max speed:
222  Eigen::Vector3d vel_result = v_sum * max_speed_;
223 
224  if (rtree_neighbors.size() > 0) {
225 
226  velocity_controller(vel_result);
227 
228  if (show_shapes_) {
229  ShapePtr sphere(new scrimmage_proto::Shape);
230  sphere->set_opacity(0.1);
231  sc::set(sphere->mutable_color(), 0, 255, 0);
232  sphere->mutable_sphere()->set_radius(sphere_of_influence_);
233  sc::set(sphere->mutable_sphere()->mutable_center(), state_->pos());
234  draw_shape(sphere);
235 
236  // Draw resultant vector:
237  ShapePtr line(new scrimmage_proto::Shape);
238  line->set_opacity(0.75);
239  sc::set(line->mutable_color(), 255, 255, 0);
240  sc::set(line->mutable_line()->mutable_start(), state_->pos());
241  sc::set(line->mutable_line()->mutable_end(), vel_result + state_->pos());
242  draw_shape(line);
243  }
244  } else {
245  velocity_controller(v_goal);
246  }
247  return true;
248 }

259

263
