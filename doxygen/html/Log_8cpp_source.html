197 bool Log::parse(std::string filename, Log::FileType type) {
198  int input_fd = open(filename.c_str(), O_RDONLY);
199  if (input_fd == -1) {
200  cout << "Failed to open file: " << filename.c_str() << endl;
201  return false;
202  }
203 
204  ZeroCopyInputStreamPtr input = std::make_shared<google::protobuf::io::FileInputStream>(input_fd);
205 
206  if (!input) {
207  cout << "Failed to open FileInputStream. Filename: "
208  << filename.c_str() << endl;
209  }
210 
211  if (type == FRAMES) {
212  parse_frames(filename, input);
213  } else if (type == SHAPES) {
214  parse_shapes(filename, input);
215  } else if (type == UTMTERRAIN) {
216  parse_utm_terrain(filename, input);
217  } else if (type == CONTACTVISUAL) {
218  parse_contact_visual(filename, input);
219  } else if (type == MSG) {
220  // parse_contact_visual(filename, input);
221  cout << "No parser for messages yet. " << type << endl;
222  } else {
223  cout << "Log parse(): Invalid parse type: " << type << endl;
224  }
225  close(input_fd);
226 
227  return true;

300 bool Log::writeDelimitedTo(const google::protobuf::MessageLite& message,
301  ZeroCopyOutputStreamPtr rawOutput) {
302  if (mode_ == READ || !enable_log_) {
303  return true;
304  }
305 
306  // We create a new coded stream for each message. Don't worry, this is fast.
307  google::protobuf::io::CodedOutputStream output(rawOutput.get());
308 
309  // Write the size.
310  const int size = message.ByteSize();
311  output.WriteVarint32(size);
312 
313  // cout << "Writing message of size: " << size << endl;
314 
315  uint8_t* buffer = output.GetDirectBufferForNBytesAndAdvance(size);
316  if (buffer != NULL) {
317  // Optimization: The message fits in one buffer, so use the faster
318  // direct-to-array serialization path.
319  message.SerializeWithCachedSizesToArray(buffer);
320  } else {
321  // Slightly-slower path when the message is multiple buffers.
322  message.SerializeWithCachedSizes(&output);
323  if (output.HadError()) {
324  return false;
325  }
326  }
327 
328  return true;
329 }

331 bool Log::readDelimitedFrom(const std::string &filename,
332  ZeroCopyInputStreamPtr rawInput,
333  MessageLitePtr message, bool& clean_eof) {
334  // see here for the implementation: http://stackoverflow.com/a/22927149
335 
336  // We create a new coded stream for each message. Don't worry, this is fast,
337  // and it makes sure the 64MB total size limit is imposed per-message rather
338  // than on the whole stream. (See the CodedInputStream interface for more
339  // info on this limit.)
340  google::protobuf::io::CodedInputStream input(rawInput.get());
341  const int start = input.CurrentPosition();
342  clean_eof = false;
343 
344  // Read the size.
345  uint32_t size;
346  if (!input.ReadVarint32(&size)) {
347  clean_eof = (input.CurrentPosition() == start);
348  return false;
349  }
350  // Tell the stream not to read beyond that size.
351  google::protobuf::io::CodedInputStream::Limit limit = input.PushLimit(size);
352 
353  // Parse the message.
354  if (!message->MergeFromCodedStream(&input)) {
355  cout << "Error: MergeFromCodedStream failed: " << filename << endl;
356  return false;
357  }
358 
359  if (!input.ConsumedEntireMessage()) {
360  cout << "Error: ConsumedEntireMessage() failed: " << filename << endl;
361  return false;
362  }
363 
364  // Release the limit.
365  input.PopLimit(limit);
366 
367  return true;
368 }
