98 bool LOSSensor::step() {
99  // Call the super class first
100  bool retVal = RayTrace::step();
101  if (retVal == false) {
102  return retVal;
103  }
104  // Check whether an update is available if not using the collision service
105  if (use_flat_earth_ == true) {
106  if (time_->t() < (last_update_time_ + max_sample_rate())) {
107  // No update, no message sent
108  return retVal;
109  }
110  } else { // Updated rate managed by collision system
111  if (subscribed_ == false) {
112  subscribed_ = true;
113  // Register for the collision callbacks. Assume set up for local network publish
114  auto pc_cb = [&] (scrimmage::MessagePtr<sensor::RayTrace::PointCloud> msg) {
115  range_ = msg->data.points[0].point.norm();
116  oor_ = msg->data.points[0].oor;
117  new_data_ = true;
118  };
119  std::string topic_name = std::to_string(parent_->id().id()) + "/" + name() + "/pointcloud";
120  subscribe<sensor::RayTrace::PointCloud>("LocalNetwork", topic_name, pc_cb);
121  printf("LOSSensor: Subscribing with topic name %s\n", topic_name.c_str());
122  }
123  if (new_data_ == false) {
124  return retVal;
125  }
126  }
127 
128  // Can update
129  last_update_time_ = time_->t();
130  new_data_ = false;
131 
132  // Coordinate systems used:
133  // Local level: East/North/Up
134  // Body: Nose/Left/Up
135 
136  bool oor = false;
137  double range = 0.0;
138  if (use_flat_earth_ == true) {
139  // Get appropriate conversion
140  Eigen::Matrix4d tf_m = parent_->state_truth()->tf_matrix(false) *
141  transform()->tf_matrix();
142  // Transform sensor's origin to world coordinates
143  Eigen::Vector4d sensor_pos = tf_m * Eigen::Vector4d(0, 0, 0, 1);
144  Eigen::Vector3d sensor_pos_w = sensor_pos.head<3>() + parent_->state_truth()->pos();
145 
146  // Transform ray's end point to world coordinates
147  Eigen::Vector4d ray = tf_m * Eigen::Vector4d(1.0, 0.0, 0.0, 1.0);
148  Eigen::Vector3d ray_w = ray.head<3>();
149 
150  // Verify that this will intersect the ground
151  if ((sensor_pos_w(2) < 0.0) || (ray_w(2) >= 0.0)) { // Out of range
152  oor = true;
153  } else { // Potentially in range
154  // Project the unit vector intersection to the ground.
155  range = sensor_pos_w(2) / (-1.0 * ray_w(2));
156  }
157  } else {
158  // Use a service to get range to collision
159  // Without a service available, this returns oor at all times
160  range = range_;
161  oor = oor_;
162  }
163 
164  if ((oor == false) && (probability_oor_error_ > 0.0)) {
165  // Check the random error to see whether it could swap in-range/out-of-range.
166  bool swap_in_out_range =
167  (static_cast<double>(std::rand()) / static_cast<double>(RAND_MAX)) < probability_oor_error_;
168 
169  if ((((range < min_range()) || (range > max_range())) && !swap_in_out_range) ||
170  (((range > min_range()) && (range < max_range())) && swap_in_out_range)) {
171  // Out of range
172  oor = true;
173  } else { // In-range
174  // Limit to the range boundaries (in case an error caused an in-range
175  // measurement when it was actually out of range)
176  range = MIN(MAX(range, min_range()), max_range());
177  }
178  }
179 
180  // Now that in-range versus out-of-range is known as well as the actual range,
181  // add error.
182  auto msg = std::make_shared<sc::Message<sm::LOSRangeMessage>>();
183  msg->data.set_sensor_id(sensor_id_);
184  if (oor == false) {
185  // Add error
186  if ((static_cast<double>(std::rand()) / static_cast<double>(RAND_MAX)) < probability_of_error_) {
187  // Add additional error
188  std::normal_distribution<double> error_dist(0.0, error_sd_);
189  double error = error_dist(generator_);
190  range = MIN(MAX(range + error, min_range()), max_range());
191  }
192 
193  // Finally, add the actual noise
194  double total_sd = range_sd_min_ + (range_sd_per_unit_ * range);
195  if (total_sd > 0.0) {
196  std::normal_distribution<double> noise_dist(0.0, total_sd);
197  range = MIN(MAX(range + noise_dist(generator_), min_range()), max_range());
198  }
199  // Set the values to the message
200  msg->data.set_range(range);
201  msg->data.set_range_var(total_sd * total_sd);
202  } else { // Out of range
203  // Set the values to the message
204  msg->data.set_range(oor_return_);
205  msg->data.set_range_var(range_sd_oor_ * range_sd_oor_);
206  }
207 
208  // Publish the message and return all good
209  los_pub_->publish(msg);
210  return retVal;
211 }
