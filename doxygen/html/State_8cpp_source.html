69 bool State::InFieldOfView(State &other, double fov_width, double fov_height) const {
70  Eigen::Vector3d rel_pos = this->rel_pos_local_frame(other.pos());
71  double az = atan2(rel_pos(1), rel_pos(0));
72  double norm_xy = sqrt(pow(rel_pos(0), 2) + pow(rel_pos(1), 2));
73  double el = atan2(rel_pos(2), norm_xy);
74  return std::abs(az) < fov_width / 2 && std::abs(el) < fov_height / 2;
75 }

99 Eigen::Matrix4d State::tf_matrix(bool enable_translate) {
100  // Construct 4x4 matrix of the form:
101  // |rot|rot|rot| sx |
