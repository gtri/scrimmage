190 boost::optional<std::string> run_test(const std::string& mission,
191  const bool& init_python,
192  const bool& shutdown_python) {
193  SimControl simcontrol;
194  if (not simcontrol.init(mission, init_python)) {
195  cout << "Failed to initialize SimControl." << endl;
196  return boost::none;
197  }
198 
199  // Handle kill signals
200  struct sigaction sa;
201  memset( &sa, 0, sizeof(sa) );
202  shutdown_handler = [&](int /*s*/){
203  cout << endl << "Exiting gracefully" << endl;
204  simcontrol.force_exit();
205  };
206  sa.sa_handler = signal_handler;
207  sigfillset(&sa.sa_mask);
208  sigaction(SIGINT, &sa, NULL);
209  sigaction(SIGTERM, &sa, NULL);
210 
211  simcontrol.mp()->set_time_warp(-1);
212  simcontrol.mp()->set_enable_gui(false);
213 
214  simcontrol.pause(false);
215  if (not simcontrol.run()) {
216  return boost::none;
217  }
218 
219  // Extract the log directory
220  auto out = boost::optional<std::string>{simcontrol.mp()->log_dir()};
221 
222  // Shutdown SimControl
223  if (not simcontrol.shutdown(shutdown_python)) {
224  return boost::none;
225  }
226  return out;
227 }
