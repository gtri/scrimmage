251 bool get_vec_of_vecs(const std::string &str,
252  std::vector<std::vector<std::string>> &out,
253  const std::string &delims) {
254  // Of the form...
255  // [a b c d]
256  // [e f g h]
257  // [h i j k]
258  out.clear();
259 
260  std::string str_copy(str);
261  remove_leading_spaces(str_copy);
262  remove_trailing_spaces(str_copy);
263 
264  // First split based on user defined delimiters (typically ", ")
265  std::vector<std::string> tokens;
266  boost::algorithm::split(tokens, str_copy, boost::is_any_of(delims),
267  boost::token_compress_on);
268 
269  // The tokens vector consists of the delimited items. Break up each vector
270  // based on the start "[" and stop "]" delimiters. However, we will allow
271  // one-level of nested start and stop delimiters.
272  std::vector<std::string> items;
273  int open_bracket_count = 0;
274  std::string item = "";
275  for (std::string token : tokens) {
276 
277  // Remove whitespace
278  token.erase(
279  std::remove_if(token.begin(),
280  token.end(),
281  [](unsigned char x){return std::isspace(x);}),
282  token.end());
283 
284  // cout << token << endl;
285 
286  for (unsigned int i = 0; i < token.size(); i++) {
287  if (token[i] == '[') {
288  if (open_bracket_count > 0) {
289  item += "[";
290  }
291  ++open_bracket_count;
292  } else if (token[i] == ']') {
293  if (open_bracket_count > 1) {
294  item += "]";
295  }
296  --open_bracket_count;
297 
298  if (open_bracket_count == 0) {
299  items.push_back(item);
300  out.push_back(items);
301  items.clear();
302  item = "";
303  }
304  } else if (open_bracket_count == 2) {
305  item += std::string(1, token[i]);
306  } else if (open_bracket_count > 0) {
307  item += std::string(1, token[i]);
308  }
309  }
310 
311  if (open_bracket_count == 1) {
312  items.push_back(item);
313  item.clear();
314  } else if (open_bracket_count == 0) {
315  out.push_back(items);
316  item.clear();
317  items.clear();
318  }
319  }
320 
321  // Remove any empty strings from the individual vectors
322  for (auto &vec : out) {
323  vec.erase(std::remove_if(vec.begin(), vec.end(), [] (auto &str) {
324  return str.empty();
325  }), vec.end());
326  }
327 
328  // Remove any empty vectors from the out vector of vectors
329  out.erase(std::remove_if(out.begin(), out.end(), [] (auto &vec) {
330  return vec.empty();
331  }), out.end());
332 
333  return true;
334 }
